45      private List<String> queryParameters() {
46          List<String> l = qi.getParameters();
47          if (aposMap != null) {
48              return parameterList;
49          }
50          aposMap = new HashMap<>();
51          int i = 0;
52          parameterList = new ArrayList<>();
53          parameterList.addAll(l);
54          for (String par : parameterList) {
55              if (par.contains("'") || par.contains("\"")) {
56                  int index = Math.max(Math.max(par.lastIndexOf(' '), par.lastIndexOf('\n')), par.lastIndexOf('\r'));
57  
58                  String decl = par.substring(0, index).trim();
59                  if (decl.startsWith("[")) {
60                      decl = decl.substring(1, decl.length() - 1);
61                  }
62                  String h = treatApos(decl);
63                  aposMap.put(h, decl);
64  
65                  parameterList.set(i, treatApos(par));
66              }
67              i++;
68          }
69          return parameterList;
70      }

72      public void createProcedure() {
73          isProcedure = true;
74          String sql = null;
75          try {
76  
77              List<String> l = queryParameters();
78              sql = getSQL();
79  
80              if (l == null || l.isEmpty()) {
81                  parametersEmpiric();
82  
83              } else {
84                  sql = SQLConverter.removeParameters(sql);
85                  parametersDeclared();
86                  if (!conversionOk) {
87                      parametersEmpiric(true);
88                  }
89              }
90  
91              if (conversionOk) {
92                  exception = null;
93  
94              } else {
95                  return;
96              }
97              String inside = convertSQL(convertApos(sql)).trim();
98              if (!inside.endsWith(";")) {
99                  inside = inside + ";";
100             }
101             String procedureName = SQLConverter.escapeIdentifier(qi.getName(), hsqldb);
102             String procedure = "CREATE PROCEDURE " + procedureName + "(" + parameters + ") MODIFIES SQL DATA \n"
103                     + " BEGIN ATOMIC " + inside + "\n END";
104 
105             if (exec(procedure)) {
106                 signature = qi.getName() + "(" + originalParameters + ")";
107                 loaded = true;
108             }
109         } catch (Exception _ex) {
110             exception = _ex;
111         }

115         String qnn = null;
116         String sql = null;
117         try {
118             List<String> l = queryParameters();
119             qnn = SQLConverter.escapeIdentifier(qi.getName(), hsqldb);
120             sql = getSQL();
121 
122             if (l == null || l.isEmpty()) {
123                 parametersEmpiric();
124 
125             } else {
126                 sql = SQLConverter.removeParameters(sql);
127                 parametersDeclared();
128                 if (!conversionOk) {
129                     parametersEmpiric();
130                 }
131             }
132             if (conversionOk) {
133                 exception = null;
134 
135             } else {
136                 return;
137             }
138 
139             String inside = convertSQL(convertApos(sql)).trim();
140             if (inside.endsWith(";")) {
141                 inside = inside.substring(0, inside.length() - 1);
142             }
143             String funName = qnn;
144             String function = "CREATE FUNCTION " + funName + "(" + parameters + ") RETURNS " + "TABLE ("
145                     + getTableDefinition() + ") READS SQL DATA \n" + " RETURN (TABLE(" + inside + "));";
146             if (!exec(function)) {
147                 return;
148             }
149             String createView = "CREATE VIEW " + qnn + " AS SELECT * FROM TABLE(" + funName + "("
150                     + defaultParameterValues + "))";
151 
152             if (!exec(createView)) {
153                 return;
154             }
155             loaded = true;
156         } catch (Exception _ex) {
157             exception = _ex;
158         }

161     private String convertApos(String sql) {
162         if (aposMap != null) {
163             for (Map.Entry<String, String> me : aposMap.entrySet()) {
164                 sql = sql.replaceAll("(?i)" + Pattern.quote("[" + me.getValue() + "]"), "[" + me.getKey() + "]");
165             }
166         }
167 
168         return sql;
169     }

171     private String getTableDefinition() throws SQLException {
172         ResultSet rs = ps.executeQuery();
173         ResultSetMetaData rsmd = rs.getMetaData();
174         StringBuilder sb = new StringBuilder();
175         String comma = "";
176         for (int i = 1; i <= rsmd.getColumnCount(); i++) {
177             String type = completeTypeName(rsmd.getColumnTypeName(i), rsmd.getPrecision(i), rsmd.getScale(i), false);
178             sb.append(comma).append(SQLConverter.escapeIdentifier(rsmd.getColumnLabel(i), hsqldb)).append(" ")
179                     .append(type);
180             comma = ",";
181         }
182         return sb.toString();
183     }

185     private String completeTypeName(String type, int i, int j, boolean useDefault) {
186         if ("VARCHAR".equals(type)) {
187             type += "(" + i + ")";
188         }
189         if ("NUMERIC".equals(type)) {
190             if (useDefault) {
191                 type += "(100,10)";
192             } else {
193                 type += "(" + i + "," + j + ")";
194             }
195         }
196         return type;
197     }

220     private static String transalteFormFields(String sqlString) {
221         return sqlString.replaceAll("\\[([^\\]]*)\\]\\!\\[([^\\]]*)\\]\\!\\[([^\\]]*)\\]", "[$1_$2_$3]")
222                 .replaceAll("\\[(\\w*)\\]\\!\\[(\\w*)\\]\\!\\[(\\w*)\\]", "[$1_$2_$3]")
223                 .replaceAll("((?i)FORMS)\\.(\\w*)\\.(\\w*)", "[$1_$2_$3]");
224     }

246     private void parametersDeclared() {
247         List<String> ls = queryParameters();
248         StringBuilder args = new StringBuilder();
249         String comma = "";
250         List<String> ar = new ArrayList<>();
251         for (String par : ls) {
252             par = par.trim();
253             int index = Math.max(Math.max(par.lastIndexOf(' '), par.lastIndexOf('\n')), par.lastIndexOf('\r'));
254 
255             String decl = par.substring(0, index).trim();
256             String type = par.substring(index).trim();
257             if (!decl.startsWith("[")) {
258                 decl = "[" + decl + "]";
259             }
260             ar.add(decl);
261             String type0 = type.indexOf('(') > 0 ? type.substring(0, type.indexOf('(')) : type;
262             String typeS = type.indexOf('(') > 0 ? type.substring(type.indexOf('(')) : "";
263             Map<String, String> hm = TypesMap.getAccess2HsqlTypesMap();
264 
265             type = hm.get(type0.toUpperCase()) + typeS;
266             if ("VARCHAR".equalsIgnoreCase(type)) {
267                 type += "(255)";
268             }
269 
270             args.append(comma).append(decl).append(" ").append(type);
271             comma = ",";
272 
273         }
274         String sql = getSQL();
275         sql = convertApos(SQLConverter.removeParameters(sql));
276 
277         for (String var : ar) {
278             sql = sql.replaceAll("(?i)" + Pattern.quote(var), "?");
279         }
280 
281         try {
282             ps = hsqldb.prepareStatement(convertSQL(sql));
283             if (!isProcedure) {
284                 ParameterMetaData pmd = ps.getParameterMetaData();
285                 StringBuilder defPar = new StringBuilder();
286                 comma = "";
287 
288                 for (int i = 1; i <= pmd.getParameterCount(); i++) {
289                     if (!isProcedure) {
290                         ps.setNull(i, pmd.getParameterType(i));
291                     }
292                     defPar.append(comma).append("NULL");
293                     comma = ",";
294                 }
295                 defaultParameterValues = defPar.toString();
296             }
297             originalParameters = args;
298             parameters = SQLConverter.convertSQL(args.toString()).getSql();
299 
300             conversionOk = true;
301         } catch (SQLException _ex) {
302             exception = _ex;
303         }
304 
305     }

322     private String convertSQL(String sql, List<String> parameters2) {
323         for (String s : parameters2) {
324             if (s.indexOf('\'') > 0 || s.indexOf('"') > 0) {
325                 String src = Pattern.quote(s);
326                 String target = treatApos(s);
327                 sql = sql.replaceAll(src, target);
328                 aposMap.put(target.substring(1, target.length() - 1), s.substring(1, s.length() - 1));
329             }
330         }
331         return convertSQL(sql);

336     private void doParametersEmpiric(Map<String, Integer> _psmp, List<String> _parameters, Map<String, String> parem, String sql) {
337         String psTxt = null;
338         try {
339             psTxt = convertSQL(sql, _parameters);
340 
341             List<String> l = SQLConverter.getParameters(psTxt);
342 
343             // apostrophe treatment
344             for (String s : l) {
345                 String h = treatApos(s);
346                 for (String modf : _parameters) {
347                     if (convertSQL(modf).equals(s)) {
348                         h = treatApos(modf);
349                         if (!aposMap.containsKey(h)) {
350                             _parameters.set(_parameters.indexOf(modf), h);
351                             psTxt = psTxt.replaceAll("(?i)" + Pattern.quote(s), convertSQL(h));
352                             sql = sql.replaceAll("(?i)" + Pattern.quote(modf), h);
353                             h = h.substring(1, h.length() - 1);
354                             aposMap.put(h, modf.substring(1, modf.length() - 1));
355                         }
356                     }
357                 }
358             }
359 
360             ps = hsqldb.prepareStatement(psTxt);
361 
362             ParameterMetaData pmd = ps.getParameterMetaData();
363             _psmp = reorderIndexes(_psmp, parem);
364             List<String> ar = new ArrayList<>(_psmp.keySet());
365             List<Integer> pI = parIndexes(sql);
366             StringBuilder parS = new StringBuilder();
367             StringBuilder defPar = new StringBuilder();
368             int j = 0;
369             String comma = "";
370             for (int i = 1; i <= pmd.getParameterCount(); i++) {
371                 if (!isProcedure) {
372                     ps.setNull(i, pmd.getParameterType(i));
373                 }
374 
375                 if (j > ar.size() - 1) {
376                     continue;
377                 }
378                 String key = ar.get(j);
379                 int index = _psmp.get(key);
380                 if (index == pI.get(i - 1)) {
381                     defPar.append(comma).append("NULL");
382                     String type =
383                             completeTypeName(pmd.getParameterTypeName(i), pmd.getPrecision(i), pmd.getScale(i), true);
384                     parS.append(comma).append(SQLConverter.escapeIdentifier(key, hsqldb)).append(" ").append(type);
385                     comma = ",";
386                     j++;
387                 }
388 
389             }
390             parameters = parS.toString();
391             defaultParameterValues = defPar.toString();
392             conversionOk = true;
393 
394         } catch (SQLException _ex) {
395 
396             for (String par : _parameters) {
397                 String par1 = SQLConverter.preEscapingIdentifier(par.substring(1, par.length() - 1));
398                 int index = sql.toUpperCase().indexOf(par.toUpperCase());
399                 if (index >= 0 && _ex.getMessage() != null && (_ex.getMessage().toUpperCase().endsWith(": " + par1)
400                         || _ex.getMessage().toUpperCase().contains(": " + par1 + " IN STATEMENT "))) {
401                     sql = sql.replaceAll("(?i)" + Pattern.quote(par), "?");
402                     _psmp.put(par1, index);
403                     parem.put(par1, par);
404                     String parname = originalParameters.length() == 0 ? par : "," + par;
405                     originalParameters.append(parname);
406                     doParametersEmpiric(_psmp, _parameters, parem, sql);
407 
408                 } else {
409                     exception = _ex;
410                 }
411             }
412             if (exception == null) {
413                 exception = _ex;
414 
415             }
416 
417         }

421     private Map<String, Integer> reorderIndexes(Map<String, Integer> psmp, Map<String, String> parem) {
422         Map<Integer, String> tm = new TreeMap<>();
423         Integer[] nI = new Integer[psmp.size()];
424         String[] sI = new String[psmp.size()];
425         Map<String, Integer> dI = new HashMap<>();
426         int j = 0;
427         for (Map.Entry<String, Integer> me : psmp.entrySet()) {
428             tm.put(me.getValue(), me.getKey());
429             nI[j] = me.getValue();
430             sI[j] = me.getKey();
431             dI.put(sI[j], 0);
432             j++;
433         }
434 
435         boolean changedSignature = false;
436         for (int i = 0; i < tm.size() - 1; i++) {
437 
438             for (j = i + 1; j < tm.size(); j++) {
439                 if (nI[j] < nI[i]) {
440                     changedSignature = true;
441                     dI.put(sI[i], dI.get(sI[i]) - parem.get(sI[j]).length() + 1);
442                 }
443             }
444         }
445 
446         Map<String, Integer> rlhm = new LinkedHashMap<>();
447         for (Map.Entry<Integer, String> me : tm.entrySet()) {
448             rlhm.put(me.getValue(), me.getKey() + dI.get(me.getValue()));
449         }
450         if (changedSignature) {
451             StringBuilder sb = new StringBuilder();
452             String comma = "";
453             for (String key : rlhm.keySet()) {
454                 sb.append(comma).append(parem.get(key));
455                 comma = ",";
456             }
457             originalParameters = sb;
458         }
459 
460         return rlhm;
461     }
