113         List<String> keywordList = new ArrayList<>(KEYWORDLIST);
114         keywordList.remove("SELECT");
115         StringBuilder keywords = new StringBuilder();
116         String sep = "";
117         for (String s : keywordList) {

124     private static void aliases(String sql, NormalizedSQL nsql) {
125         Matcher mtc = SELECT_FROM_PATTERN_START.matcher(sql);
126         if (mtc.find()) {
127 
128             int init = mtc.end();
129             sql = sql.substring(init);
130             mtc = SELECT_FROM_PATTERN_END.matcher(sql);
131             if (mtc.find()) {
132                 int end = mtc.start();
133                 sql = sql.substring(0, end);
134 
135                 for (mtc = UNESCAPED_ALIAS.matcher(sql); mtc.find(); mtc = UNESCAPED_ALIAS.matcher(sql)) {
136                     int e = mtc.end();
137 
138                     sql = sql.substring(e) + " ";
139                     char[] sqlc = sql.toCharArray();
140                     if (sqlc[0] == '[') {
141                         continue;
142                     }
143                     StringBuilder sb = new StringBuilder();
144                     for (char c : sqlc) {
145                         if (c == ' ' || c == '\n' || c == '\r' || c == ',') {
146                             String key = SQLConverter.preEscapingIdentifier(sb.toString());
147                             nsql.put(key, sb.toString());
148                             break;
149                         } else {
150                             sb.append(c);
151                         }
152                     }
153                 }
154 
155             }
156         }
157     }

174     private static int[] getQuoteGroup(String _s) {
175         if (!_s.contains("''")) {
176             Matcher mtc = QUOTE_M_PATTERN.matcher(_s);
177             return mtc.find() ? new int[] {mtc.start(), mtc.end()} : null;
178         } else {
179             int[] ret = new int[] {-1, -1};
180             Matcher mc = QUOTE_S_PATTERN.matcher(_s);
181             while (mc.find()) {
182                 int start = mc.start();
183                 int end = mc.end();
184                 if ((end - start) % 2 == 0) {
185                     if (ret[0] == -1) {
186                         return new int[] {mc.start(), mc.end()};
187                     }
188                 } else {
189                     if (ret[0] == -1) {
190                         ret[0] = mc.start();
191                     } else {
192                         ret[1] = mc.end();
193                         return ret;
194                     }
195                 }
196             }
197             return null;
198         }
199     }

395     public static NormalizedSQL convertSQL(String sql, UcanaccessConnection conn, boolean creatingQuery) {
396         NormalizedSQL nsql = new NormalizedSQL();
397         sql = sql + " ";
398         aliases(sql, nsql);
399         sql = replaceBacktrik(sql);
400         sql = replaceAposNames(sql);

402         sql = convertAccessDate(sql);
403         sql = convertQuotedAliases(sql, nsql);
404         sql = escape(sql);
405         sql = convertLike(sql);
406         sql = replaceWhiteSpacedTables(sql);
407         // sql = replaceExclamationPoints(sql);
408         if (!creatingQuery) {
409             Pivot.checkAndRefreshPivot(sql, conn);
410             sql = DFunction.convertDFunctions(sql, conn);
411         }
412         sql = sql.trim();
413 
414         nsql.setSql(sql);
415 
416         return nsql;
417     }

446     private static String convertQuotedAliases(String sql, NormalizedSQL nsql) {
447         for (Matcher mtc = KIND_OF_SUBQUERY.matcher(sql); mtc.find(); mtc = KIND_OF_SUBQUERY.matcher(sql)) {
448             String g2 = mtc.group(2).trim();
449             if (g2.endsWith(";")) {
450                 g2 = g2.substring(0, g2.length() - 1);
451             }
452             sql = sql.substring(0, mtc.start()) + "(" + g2 + ")" + sql.substring(mtc.end());
453         }
454         Set<String> hs = new HashSet<>();
455         String sqle = sql;
456         String sqlN = "";
457         for (Matcher mtc = QUOTED_ALIAS.matcher(sqle); mtc.find(); mtc = QUOTED_ALIAS.matcher(sqle)) {
458             String g2 = mtc.group(2);
459             if (g2.indexOf('\'') >= 0 || g2.indexOf('"') >= 0) {
460                 hs.add(g2);
461             }

994     private static int secondOperandIndex(String sql, int i) {
995         sql = sql.substring(i + 1);
996         char[] ca = sql.toCharArray();
997         boolean foundType = false;
998         boolean field = false;
999         boolean group = false;
1000         boolean digit = false;
1001         int countPar = 0;
1002         int j;
1003 
1004         for (j = 0; j < ca.length; j++) {
1005             char c = ca[j];
1006             if (c == ' ') {
1007                 continue;
1008             }
1009 
1010             if (foundType) {
1011                 if (field && c == ']') {
1012                     return j;
1013                 } else if (digit && !Character.isDigit(c) && c != '.') {
1014                     return j - 1;
1015                 } else if (group) {
1016                     if (c == '(') {
1017                         countPar++;
1018                     }
1019                     if (c == ')') {
1020                         countPar--;
1021                         if (countPar == 0) {
1022                             return j;
1023                         }
1024                     }
1025                 }
1026             } else {
1027                 if (c == '[') {
1028                     foundType = true;
1029                     field = true;
1030                 } else if (c == '(') {
1031                     foundType = true;
1032                     group = true;
1033                     countPar++;
1034                 } else if (Character.isDigit(c)) {
1035                     foundType = true;
1036                     digit = true;
1037                 } else if (c == '+' || c == '-') {
1038                     if (j + 1 < ca.length && ca[j + 1] != '(' && ca[j + 1] != '[') {
1039                         foundType = true;
1040                         digit = true;
1041                     }
1042                 }
1043 
1044             }
1045 
1046         }
1047         return j - 1;
1048     }

1058         int countPar = 0;

1060         for (j = ca.length - 1; j >= 0; j--) {
1061             char c = ca[j];
1062 
1063             if (c == ' ') {
1064                 continue;
1065             }
1066             if (foundType) {
1067                 if (field && c == '[') {
1068                     return j;
1069                 } else if (digit && !Character.isDigit(c) && c != '.') {
1070                     if ((c == '+' || c == '-') && j > 0 && (ca[j - 1] == '+' || ca[j - 1] == '-')) {
1071                         return j;
1072                     }
1073 
1074                     return j + 1;
1075                 } else if (group) {
1076                     if (c == ')') {
1077                         countPar++;
1078                     }
1079 
1080                     if (c == '(') {
1081                         countPar--;
1082                         if (countPar == 0) {
1083                             return j;
1084                         }
1085                     }
1086                 }
1087             } else {
1088                 if (c == ']') {
1089                     foundType = true;
1090                     field = true;
1091                 } else if (c == ')') {
1092                     foundType = true;
1093                     group = true;
1094                     countPar++;
1095                 } else if (Character.isDigit(c)) {
1096                     foundType = true;
1097                     digit = true;
1098                 }
1099 
1100             }
1101 
1102         }
1103         return j + 1;
1104     }
